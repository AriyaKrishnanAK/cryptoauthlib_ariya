57 // Now we use the generic helper definitions above to define PKCS11_API and PKCS11_LOCAL.
58 // PKCS11_API is used for the public API symbols. It either DLL imports or DLL exports (or does nothing for static build)
59 // PKCS11_LOCAL is used for non-api symbols.
60 
61 #ifdef PKCS11_DLL_EXPORTS // defined if we are building the PKCS11 DLL (instead of using it)
62 #define PKCS11_API PKCS11_HELPER_DLL_EXPORT
63 #elif PKCS11_DLL // defined if PKCS11 is compiled as a DLL
64 #define PKCS11_API PKCS11_HELPER_DLL_IMPORT
65 #else // PKCS11_DLL is not defined: this means PKCS11 is a static lib.
66 #define PKCS11_API
67 #endif // PKCS11_DLL
68 #define PKCS11_LOCAL PKCS11_HELPER_DLL_LOCAL
